\par\noindent We want to come up with a charecterization of planar graphs. We proved $K_{5}$ and $K_{3,3}$ are nonplanar.\n

\defn{$H$ is a \underline{minor} of $G$ if it can be obtained from $G$ by deleting vertexes, deleting edges, and contracting edges.}

\defn{$H$ is a \underline{subdivision} of $G$ if it can be obtained by adding vertexes to split edges, removing edges, and removing vertexes.}

\thm{(Wagner) A graph $G$ is nonplanar iff $G$ contains $K_{5}$ or $K_{3,3}$ as a minor.}

\thm{(Kuratowski) A graph $G$ is nonplanar iff $G$ containes $K_{5}$ or $K_{3,3}$ as a subdivision.\n
Proof:\n
Claim 1: If $G$ has no subdivision of $K_{5}$ or $K_{3,3}$ and $e\in{}G$, then $G/e=H$ has no subdivision of $K_{5}$ or $K_{3,3}$. If $H'\subset{}H$ has a subdivision of $K_{5}$ or $K_{3,3}$, and $z\in{}H'$, then $H'$ is a subdivision of $G$. If $z\in{}H'$, and $\ptxt{deg}_{H'}(z)=2$ (then just do every case) \flag{I'm not sure what he means here}\n
Claim 2: (Inductive-ness) let $G'=G/e$. Then $G'$ has fewer edges than $G$. By induction, $G'$ is planar.\proven}

\par\noindent A torus is a doughnut. We can draw graphs onto a torus.\n

\ex{$K_{7}$ can be drawn on a torus.}

\defn{$T_{g}$ is the sphere with $g$ handles.}

\defn{Given $G$ and $T_{g}$, an \underline{embedding} of $G$ on $T_{g}$ is a drawing of $G$ on $T_{g}$ \st{} no $2$ edges cross.}

\defn{Given $G$ and $T_{g}$, a \underline{$2$-cell embedding} of $G$ on $T_{g}$ is a drawing such that no $2$ edges cross and\n $T_{g}\setminus(V\cup{}E)=\sqcup{}D_{i}$ where each $D_{i}$ is a ``disc''.}

\thm{If $G$ has an embedding on $T_{g}$, then $v-e+f\ge{}2-2g$.\n
If $G$ has a $2$-cell embedding on $T_{g}$, then $v-e+f=2-2g$.}
